在Python中，可以使用多种方式遍历指定文件夹中的图片文件。以下是几种常见的实现方法：

# 方法1：使用os模块

import os

def traverse_images_os(folder_path):
    """使用os模块遍历图片"""
    image_extensions = ['.jpg', '.jpeg', '.png', '.gif', '.bmp', '.tiff', '.webp']
    image_files = []
    
    for filename in os.listdir(folder_path):
        # 获取文件扩展名并转换为小写
        ext = os.path.splitext(filename)[1].lower()
        if ext in image_extensions:
            full_path = os.path.join(folder_path, filename)
            image_files.append(full_path)
    
    return image_files

# 使用示例
folder = "your/image/folder/path"
images = traverse_images_os(folder)
for img in images:
    print(img)

# 方法2：使用pathlib模块

from pathlib import Path

def traverse_images_pathlib(folder_path):
    """使用pathlib模块遍历图片（Python 3.4+）"""
    image_extensions = {'.jpg', '.jpeg', '.png', '.gif', '.bmp', '.tiff', '.webp'}
    folder = Path(folder_path)
    
    # 使用列表推导式
    image_files = [
        str(file) for file in folder.iterdir() 
        if file.is_file() and file.suffix.lower() in image_extensions
    ]
    
    return image_files

# 使用示例
folder = "your/image/folder/path"
images = traverse_images_pathlib(folder)
for img in images:
    print(img)

# 方法3：递归遍历子文件夹

import os
from pathlib import Path

def traverse_images_recursive(folder_path):
    """递归遍历文件夹及其所有子文件夹中的图片"""
    image_extensions = {'.jpg', '.jpeg', '.png', '.gif', '.bmp', '.tiff', '.webp'}
    folder = Path(folder_path)
    image_files = []
    
    # 使用rglob递归遍历
    for file_path in folder.rglob('*'):
        if file_path.is_file() and file_path.suffix.lower() in image_extensions:
            image_files.append(str(file_path))
    
    return image_files

# 使用示例
folder = "your/image/folder/path"
images = traverse_images_recursive(folder)
for img in images:
    print(img)

# 方法4：使用glob模块

import glob
import os

def traverse_images_glob(folder_path):
    """使用glob模块遍历图片"""
    # 支持的图片格式模式
    patterns = [
        '*.jpg', '*.jpeg', '*.png', '*.gif', 
        '*.bmp', '*.tiff', '*.webp'
    ]
    
    image_files = []
    for pattern in patterns:
        # 搜索所有匹配的文件
        search_pattern = os.path.join(folder_path, pattern)
        image_files.extend(glob.glob(search_pattern))
        # 同时搜索大写扩展名
        search_pattern_upper = os.path.join(folder_path, pattern.upper())
        image_files.extend(glob.glob(search_pattern_upper))
    
    return list(set(image_files))  # 去重

# 使用示例
folder = "your/image/folder/path"
images = traverse_images_glob(folder)
for img in images:
    print(img)

# 方法5：完整的图片遍历类

import os
from pathlib import Path
from typing import List, Generator

class ImageTraverser:
    """图片遍历器类"""
    
    def __init__(self, supported_extensions=None):
        self.supported_extensions = supported_extensions or [
            '.jpg', '.jpeg', '.png', '.gif', '.bmp', '.tiff', '.webp'
        ]
    
    def traverse(self, folder_path: str, recursive: bool = False) -> List[str]:
        """遍历文件夹中的图片文件"""
        folder = Path(folder_path)
        
        if not folder.exists():
            raise ValueError(f"文件夹不存在: {folder_path}")
        
        if recursive:
            # 递归遍历
            image_files = [
                str(file) for file in folder.rglob('*')
                if file.is_file() and file.suffix.lower() in self.supported_extensions
            ]
        else:
            # 非递归遍历
            image_files = [
                str(file) for file in folder.iterdir()
                if file.is_file() and file.suffix.lower() in self.supported_extensions
            ]
        
        return image_files
    
    def traverse_generator(self, folder_path: str, recursive: bool = False) -> Generator[str, None, None]:
        """生成器方式遍历图片（节省内存）"""
        folder = Path(folder_path)
        
        if not folder.exists():
            raise ValueError(f"文件夹不存在: {folder_path}")
        
        if recursive:
            iterator = folder.rglob('*')
        else:
            iterator = folder.iterdir()
        
        for file_path in iterator:
            if file_path.is_file() and file_path.suffix.lower() in self.supported_extensions:
                yield str(file_path)

# 使用示例
traverser = ImageTraverser()

# 非递归遍历
images = traverser.traverse("your/image/folder/path")
for img in images:
    print(img)

# 递归遍历
images_recursive = traverser.traverse("your/image/folder/path", recursive=True)
for img in images_recursive:
    print(img)

# 使用生成器（节省内存）
for img_path in traverser.traverse_generator("your/image/folder/path", recursive=True):
    print(img_path)

# 方法6：带文件信息的高级遍历

import os
from pathlib import Path
from datetime import datetime

def traverse_images_with_info(folder_path, recursive=False):
    """遍历图片并返回详细信息"""
    image_extensions = {'.jpg', '.jpeg', '.png', '.gif', '.bmp', '.tiff', '.webp'}
    folder = Path(folder_path)
    
    image_info = []
    
    if recursive:
        file_iterator = folder.rglob('*')
    else:
        file_iterator = folder.iterdir()
    
    for file_path in file_iterator:
        if file_path.is_file() and file_path.suffix.lower() in image_extensions:
            stat = file_path.stat()
            info = {
                'path': str(file_path),
                'name': file_path.name,
                'size': stat.st_size,  # 文件大小（字节）
                'created': datetime.fromtimestamp(stat.st_ctime),  # 创建时间
                'modified': datetime.fromtimestamp(stat.st_mtime),  # 修改时间
                'extension': file_path.suffix.lower()
            }
            image_info.append(info)
    
    return image_info

# 使用示例
folder = "your/image/folder/path"
images_info = traverse_images_with_info(folder, recursive=True)

for info in images_info:
    print(f"文件名: {info['name']}")
    print(f"路径: {info['path']}")
    print(f"大小: {info['size']} 字节")
    print(f"修改时间: {info['modified']}")
    print("-" * 50)

# 使用建议

  1. 推荐使用方法2（pathlib）：代码更简洁，跨平台兼容性好
  2. 需要递归遍历时使用方法3或5：可以处理嵌套的文件夹结构
  3. 处理大量文件时使用方法5的生成器版本：节省内存
  4. 需要文件详细信息时使用方法6：获取大小、时间等元数据
